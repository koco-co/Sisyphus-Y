   ```typescript
   'use client'

   import { useState, useEffect } from 'react'
   import { X, ChevronRight, ChevronLeft } from 'lucide-react'
   import { Button } from '@/components/ui/button'

   const ONBOARDING_KEY = 'sisyphus-y-onboarding-seen'

   const STEPS = [
     {
       title: '欢迎使用 Sisyphus 测试平台',
       description: 'AI 驱动的企业级功能测试用例自动生成平台',
       content: '需求录入',
     },
     {
       title: 'AI 智能分析',
       description: '上传需求文档，AI 自动识别风险点和测试场景',
       content: 'AI 分析',
     },
     {
       title: '测试点确认',
       description: '确认 AI 生成的测试点，确保覆盖全面',
       content: '测试点确认',
     },
     {
       title: '用例生成',
       description: '基于测试点自动生成高质量测试用例',
       content: '用例生成',
     },
     {
       title: '用例管理',
       description: '统一管理所有测试用例，支持导入导出',
       content: '用例库',
     },
   ]

   export function OnboardingGuide() {
     const [show, setShow] = useState(false)
     const [step, setStep] = useState(0)

     useEffect(() => {
       const seen = localStorage.getItem(ONBOARDING_KEY)
       if (!seen) {
         setShow(true)
       }
     }, [])

     const handleClose = () => {
       localStorage.setItem(ONBOARDING_KEY, 'true')
       setShow(false)
     }

     const handleNext = () => {
       if (step < STEPS.length - 1) {
         setStep(step + 1)
       } else {
         handleClose()
       }
     }

     const handlePrev = () => {
       if (step > 0) {
         setStep(step - 1)
       }
     }

     if (!show) return null

     return (
       <div className="fixed inset-0 z-50 flex items-center justify-center bg-black/80 backdrop-blur-sm">
         <div className="relative w-full max-w-2xl bg-sy-bg-1 border border-sy-border rounded-xl p-8">
           {/* Close button */}
           <button
             onClick={handleClose}
             className="absolute top-4 right-4 text-sy-text-3 hover:text-sy-text"
           >
             <X className="w-5 h-5" />
           </button>

           {/* Step indicator */}
           <div className="flex justify-center gap-2 mb-8">
             {STEPS.map((_, i) => (
               <div
                 key={i}
                 className={`w-2 h-2 rounded-full ${
                   i === step ? 'bg-sy-accent' : 'bg-sy-border'
                 }`}
               />
             ))}
           </div>

           {/* Content */}
           <div className="text-center">
             <h2 className="text-2xl font-display text-sy-text mb-4">
               {STEPS[step].title}
             </h2>
             <p className="text-sy-text-2 mb-6">{STEPS[step].description}</p>
             <div className="bg-sy-bg-2 rounded-lg p-6 mb-8">
               <div className="text-sy-accent font-mono text-lg">
                 需求录入 → AI 分析 → 测试点确认 → 用例生成 → 用例库
               </div>
             </div>
           </div>

           {/* Navigation */}
           <div className="flex justify-between">
             <Button
               variant="ghost"
               onClick={step === 0 ? handleClose : handlePrev}
               className="text-sy-text-3"
             >
               {step === 0 ? '跳过' : '上一步'}
             </Button>
             <Button
               onClick={handleNext}
               className="bg-sy-accent hover:bg-sy-accent-2"
             >
               {step === STEPS.length - 1 ? '开始使用' : '下一步'}
               {step < STEPS.length - 1 && <ChevronRight className="w-4 h-4 ml-1" />}
             </Button>
           </div>
         </div>
       </div>
     )
   }
   ```

2. 创建 HelpFab.tsx：
   ```typescript
   'use client'

   import { useState } from 'react'
   import { HelpCircle, RotateCcw, Keyboard, MessageCircle } from 'lucide-react'
   import { cn } from '@/lib/utils'

   interface HelpFabProps {
     onShowOnboarding?: () => void
   }

   export function HelpFab({ onShowOnboarding }: HelpFabProps) {
     const [isOpen, setIsOpen] = useState(false)

     const menuItems = [
       {
         icon: RotateCcw,
         label: '重新查看引导',
         onClick: () => {
           localStorage.removeItem('sisyphus-y-onboarding-seen')
           onShowOnboarding?.()
           setIsOpen(false)
         },
       },
       {
         icon: Keyboard,
         label: '快捷键',
         onClick: () => {
           // TODO: 显示快捷键弹窗
           setIsOpen(false)
         },
       },
       {
         icon: MessageCircle,
         label: '反馈问题',
         onClick: () => {
           window.open('https://github.com/issue/new', '_blank')
           setIsOpen(false)
         },
       },
     ]

     return (
       <div className="fixed bottom-6 right-6 z-40">
         {/* Menu */}
         {isOpen && (
           <div className="absolute bottom-14 right-0 bg-sy-bg-1 border border-sy-border rounded-lg shadow-lg overflow-hidden min-w-[160px]">
             {menuItems.map((item, index) => (
               <button
                 key={index}
                 onClick={item.onClick}
                 className="flex items-center gap-3 w-full px-4 py-3 text-sy-text hover:bg-sy-bg-2 transition-colors"
               >
                 <item.icon className="w-4 h-4 text-sy-text-2" />
                 <span>{item.label}</span>
               </button>
             ))}
           </div>
         )}

         {/* FAB button */}
         <button
           onClick={() => setIsOpen(!isOpen)}
           className={cn(
             'w-12 h-12 rounded-full flex items-center justify-center transition-all',
             'bg-sy-accent hover:bg-sy-accent-2 text-white shadow-lg',
             isOpen && 'rotate-45'
           )}
         >
           <HelpCircle className="w-6 h-6" />
         </button>
       </div>
     )
   }
   ```
