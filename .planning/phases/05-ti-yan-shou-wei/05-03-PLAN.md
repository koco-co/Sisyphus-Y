   ```typescript
   function parsePromptMarkdown(content: string): PromptImport[] {
     const sections = content.split(/^---$/m).filter(Boolean)
     const prompts: PromptImport[] = []

     for (const section of sections) {
       // 匹配 ## 模块名 (module_key) 格式
       const headerMatch = section.match(/## .+ \((\w+)\)/)
       if (headerMatch) {
         const module = headerMatch[1]
         // 提取各字段内容
         const identity = extractField(section, '身份声明')
         const taskBoundary = extractField(section, '任务边界')
         const outputFormat = extractField(section, '输出规范')
         const qualityConstraints = extractField(section, '质量红线')

         prompts.push({ module, identity, taskBoundary, outputFormat, qualityConstraints })
       }
     }
     return prompts
   }
   ```
