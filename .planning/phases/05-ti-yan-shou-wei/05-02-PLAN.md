2. 恢复逻辑完善：
   ```typescript
   const handleRestore = async (ids: string[]) => {
     if (ids.length === 0 || acting) return
     setActing(true)
     try {
       const targets = filtered.filter((item) => ids.includes(item.id))
       const results = []

       for (const target of targets) {
         try {
           await recycleApi.restore(target.entity_type, target.id)
           results.push({ success: true, name: target.name })
         } catch (err) {
           // 检查是否是目录不存在错误
           if (err.message?.includes('folder not found')) {
             // 移入未分类
             await recycleApi.restoreToDefault(target.entity_type, target.id)
             toast.success(`${target.name} 已恢复到「未分类」`)
           } else {
             throw err
           }
         }
       }

       if (results.length > 0) {
         toast.success(`已恢复 ${results.length} 项`)
       }
       await loadItems()
     } catch (err) {
       toast.error(getErrorMessage(err))
     } finally {
       setActing(false)
     }
   }
   ```
