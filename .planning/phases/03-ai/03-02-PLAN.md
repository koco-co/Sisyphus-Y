```typescript
function renderMarkdown(text: string): string {
  return text
    .replace(/### (.+)/g, '<h3 class="...">$1</h3>')
    .replace(/## (.+)/g, '<h2 class="...">$1</h2>')
    .replace(/\*\*(.+?)\*\*/g, '<strong class="...">$1</strong>')
    .replace(/`([^`]+)`/g, '<code class="...">$1</code>')
    .replace(/^- (.+)$/gm, '<li class="...">• $1</li>')
    .replace(/\n\n/g, '<br/><br/>')
    .replace(/\n/g, '<br/>');  // <-- 关键：\n 已被转换为 <br/>
}
```
