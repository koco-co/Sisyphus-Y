    ```tsx
    interface WorkbenchStepBarProps {
      currentStep: 1 | 2;
      onStepClick: (step: 1 | 2) => void;
      step2Completed: boolean;  // Step2 完成后允许点击 Step1 回溯
    }

    export default function WorkbenchStepBar({ currentStep, onStepClick, step2Completed }: WorkbenchStepBarProps) {
      const steps = [
        { label: "Step 1 确认测试点", status: currentStep === 1 ? "active" : "done" },
        { label: "Step 2 生成用例", status: currentStep === 2 ? "active" : step2Completed ? "done" : "pending" },
      ];
      return (
        <div className="sticky top-0 z-10 bg-sy-bg-1 border-b border-sy-border px-4 py-2">
          <ProgressSteps steps={steps} onStepClick={(idx) => {
            // Step2 完成后允许点击 Step1 回溯
            if (idx === 0 && (currentStep === 2 || step2Completed)) onStepClick(1);
            if (idx === 1 && step2Completed) onStepClick(2);
          }} />
        </div>
      );
    }
    ```
