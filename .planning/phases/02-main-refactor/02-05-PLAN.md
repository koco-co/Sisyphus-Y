         const handleStartGenerate = () => {
           const isFirstGenerate = lastGeneratedPointIds.size === 0;
           if (isFirstGenerate) {
             // 首次生成：快照当前全量 checkedPointIds
             setLastGeneratedPointIds(new Set(checkedPointIds));
